1. my-app/pages/user-posts.tsx: This file will contain the React component for the user's posts view. It should display the post title and body, and fetch data from the API route.

```tsx
import { useEffect, useState } from 'react';
import '../styles/user-posts.scss';

type Post = {
  title: string;
  body: string;
};

export default function UserPosts() {
  const [posts, setPosts] = useState<Post[]>([]);

  useEffect(() => {
    fetch('/api/user-posts')
      .then((response) => response.json())
      .then((data) => setPosts(data));
  }, []);

  return (
    <div>
      {posts.map((post, index) => (
        <div key={index}>
          <h2>{post.title}</h2>
          <p>{post.body}</p>
        </div>
      ))}
    </div>
  );
}
```

```ts
import type { NextApiRequest, NextApiResponse } from 'next';

type Post = {
  title: string;
  body: string;
};

const dummyData: Post[] = [
  {
    title: 'Cheese Post 1',
    body: 'This is the body of the first cheese post.',
  },
  {
    title: 'Cheese Post 2',
    body: 'This is the body of the second cheese post.',
  },
];

export default function handler(req: NextApiRequest, res: NextApiResponse<Post[]>) {
  res.status(200).json(dummyData);
}
```
